# TASK-001: PDF Download Service — jsPDF Client-Side Generation with HIPAA-Compliant PHI Scoping

> **Story:** US-054 | **Epic:** EP-010 | **Sprint:** 2 | **Layer:** Frontend / Angular | **Est:** 3 h
> **Status:** Draft | **Date:** 2026-07-17

---

## Context

US-054 requires patients to download discharge instructions as a formatted PDF. The PDF is generated
entirely client-side (no server round-trip) using `jsPDF` 2.x with `jsPDF.autoTable` for structured
section layout.

PHI exposure in the PDF is governed by HIPAA minimum-necessary: only `first_name`, `discharge_date`,
and `hospital_name` are included. Full name, DOB, and MRN must never appear in the PDF output.

The PDF filename convention is:
`SmartHandoff_Discharge_Instructions_{first_name}_{discharge_date}.pdf`

The footer on every page reads:
> "Generated by SmartHandoff — Always contact your care team with questions."

**Design references:**
- US-054 Technical Notes — `jsPDF` 2.x; `jsPDF.autoTable` layout; HIPAA minimum-necessary
- US-054 Scenario 1 — formatted PDF with patient first name, discharge date, hospital name, all
  instruction sections, footer text
- design.md §3.4 — `features/patient-portal/` lazy-loaded module; `core/` for shared services
- design.md §4.1 — Angular 17; strict TypeScript mode

```typescript
/**
 * PdfDownloadService — generates HIPAA-compliant discharge instruction PDFs
 * client-side using jsPDF autoTable (US-054).
 *
 * PHI inclusion rules (HIPAA minimum-necessary):
 *   INCLUDED   : first_name, discharge_date, hospital_name
 *   EXCLUDED   : last_name, DOB, MRN, phone, email
 *
 * PDF structure per page:
 *   Header : SmartHandoff logo text + hospital_name + discharge_date
 *   Body   : one autoTable section per instruction category
 *   Footer : "Generated by SmartHandoff — Always contact your care team with questions."
 *
 * Design refs:
 *   US-054 Technical Notes  — jsPDF 2.x, HIPAA minimum-necessary
 *   US-054 Scenario 1       — formatted layout, heading per section, footer text
 *   design.md §3.4          — patient-portal feature module boundary
 */
import { Injectable } from '@angular/core';
import jsPDF from 'jspdf';
import autoTable from 'jspdf-autotable';
import { InstructionContent } from './discharge-instructions.types';

export interface PdfContext {
  /** Patient first name only — HIPAA minimum-necessary. */
  firstName: string;
  /** ISO-8601 discharge date string e.g. "2026-07-14". */
  dischargeDate: string;
  hospitalName: string;
  content: InstructionContent;
}

@Injectable({ providedIn: 'root' })
export class PdfDownloadService {
  private readonly FOOTER_TEXT =
    'Generated by SmartHandoff — Always contact your care team with questions.';

  /**
   * Generates and triggers a browser download of the discharge instructions PDF.
   *
   * @param ctx Patient context object with minimum PHI + instruction content.
   */
  download(ctx: PdfContext): void {
    const doc = new jsPDF({ orientation: 'portrait', unit: 'mm', format: 'a4' });
    const pageWidth = doc.internal.pageSize.getWidth();
    const pageHeight = doc.internal.pageSize.getHeight();
    const margin = 14;

    this.renderHeader(doc, ctx, pageWidth, margin);
    this.renderSections(doc, ctx.content, margin, pageWidth);
    this.renderFooterOnAllPages(doc, pageWidth, pageHeight);

    const safeName = ctx.firstName.replace(/\s+/g, '_');
    const safeDate = ctx.dischargeDate.replace(/-/g, '');
    doc.save(`SmartHandoff_Discharge_Instructions_${safeName}_${safeDate}.pdf`);
  }

  // ─── Private helpers ────────────────────────────────────────────────────────

  private renderHeader(
    doc: jsPDF,
    ctx: PdfContext,
    pageWidth: number,
    margin: number,
  ): void {
    // Brand header
    doc.setFontSize(18);
    doc.setFont('helvetica', 'bold');
    doc.text('SmartHandoff', margin, 20);

    // Subtitle: hospital + date
    doc.setFontSize(11);
    doc.setFont('helvetica', 'normal');
    doc.text(ctx.hospitalName, margin, 28);
    doc.text(`Discharge Date: ${ctx.dischargeDate}`, margin, 34);

    // Title line
    doc.setFontSize(14);
    doc.setFont('helvetica', 'bold');
    doc.text(
      `Discharge Instructions — ${ctx.firstName}`,
      pageWidth / 2,
      44,
      { align: 'center' },
    );

    // Horizontal rule
    doc.setDrawColor(180, 180, 180);
    doc.line(margin, 48, pageWidth - margin, 48);
  }

  private renderSections(
    doc: jsPDF,
    content: InstructionContent,
    margin: number,
    pageWidth: number,
  ): void {
    let startY = 54;

    // Medications table
    if (content.medications?.length) {
      startY = this.renderSectionHeading(doc, 'Medications', margin, startY);
      autoTable(doc, {
        startY,
        margin: { left: margin, right: margin },
        head: [['Medication', 'Dose', 'Frequency', 'Instructions']],
        body: content.medications.map((m) => [
          m.name,
          m.dose,
          m.frequency,
          m.instructions ?? '',
        ]),
        styles: { fontSize: 9 },
        headStyles: { fillColor: [41, 128, 185] },
        didDrawPage: () => {}, // footer handled separately
      });
      startY = (doc as any).lastAutoTable.finalY + 6;
    }

    // Activity
    if (content.activity) {
      startY = this.renderSectionHeading(doc, 'Activity', margin, startY);
      autoTable(doc, {
        startY,
        margin: { left: margin, right: margin },
        body: [[content.activity]],
        styles: { fontSize: 9 },
      });
      startY = (doc as any).lastAutoTable.finalY + 6;
    }

    // Diet
    if (content.diet) {
      startY = this.renderSectionHeading(doc, 'Diet', margin, startY);
      autoTable(doc, {
        startY,
        margin: { left: margin, right: margin },
        body: [[content.diet]],
        styles: { fontSize: 9 },
      });
      startY = (doc as any).lastAutoTable.finalY + 6;
    }

    // Follow-up appointments
    if (content.follow_up?.length) {
      startY = this.renderSectionHeading(doc, 'Follow-Up Appointments', margin, startY);
      autoTable(doc, {
        startY,
        margin: { left: margin, right: margin },
        head: [['Provider', 'Specialty', 'Date', 'Notes']],
        body: content.follow_up.map((f) => [
          f.provider,
          f.specialty ?? '',
          f.date ?? '',
          f.notes ?? '',
        ]),
        styles: { fontSize: 9 },
        headStyles: { fillColor: [41, 128, 185] },
      });
      startY = (doc as any).lastAutoTable.finalY + 6;
    }

    // Warning signs
    if (content.warning_signs?.length) {
      startY = this.renderSectionHeading(doc, 'Warning Signs — When to Seek Help', margin, startY);
      autoTable(doc, {
        startY,
        margin: { left: margin, right: margin },
        body: content.warning_signs.map((w) => [w]),
        styles: { fontSize: 9, textColor: [192, 0, 0] },
      });
    }
  }

  private renderSectionHeading(
    doc: jsPDF,
    title: string,
    margin: number,
    y: number,
  ): number {
    // Page overflow guard — add new page if <30 mm remaining
    const pageHeight = doc.internal.pageSize.getHeight();
    if (y + 30 > pageHeight - 15) {
      doc.addPage();
      y = 20;
    }
    doc.setFontSize(12);
    doc.setFont('helvetica', 'bold');
    doc.text(title, margin, y);
    return y + 6;
  }

  private renderFooterOnAllPages(
    doc: jsPDF,
    pageWidth: number,
    pageHeight: number,
  ): void {
    const totalPages = (doc.internal as any).getNumberOfPages();
    for (let i = 1; i <= totalPages; i++) {
      doc.setPage(i);
      doc.setFontSize(8);
      doc.setFont('helvetica', 'italic');
      doc.setTextColor(120, 120, 120);
      doc.text(this.FOOTER_TEXT, pageWidth / 2, pageHeight - 8, { align: 'center' });
      doc.setTextColor(0, 0, 0);
    }
  }
}
```

### 5. Wire handler in `DischargeInstructionsComponent`

```typescript
// Inject in the component class
private readonly pdfService = inject(PdfDownloadService);

onDownloadPdf(): void {
  const ctx = this.encounterSummary(); // signal value from TASK-002 of US-053
  if (!ctx) return;
  this.pdfService.download({
    firstName: ctx.patientFirstName,
    dischargeDate: ctx.dischargeDate,
    hospitalName: ctx.hospitalName,
    content: this.currentContent(), // computed signal from LanguageSwitcherService
  });
}
```
