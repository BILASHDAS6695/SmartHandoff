# TASK-004: Alembic Migration — Extend pharmacist_alerts Table for HIGH_RISK_DRUG_CLASS

> **Story:** US-032 | **Epic:** EP-005 | **Sprint:** 2 | **Layer:** Backend / Database | **Est:** 1 h
> **Status:** Draft | **Date:** 2026-07-16

---

## Context

TASK-003 added new columns and enum values to the `PharmacistAlert` ORM model. This task creates the corresponding Alembic migration to apply those schema changes to Cloud SQL PostgreSQL without dropping or recreating the existing `pharmacist_alerts` table.

The migration must:
1. Add the `ALTER TYPE` statements to extend the `alert_type_enum` PostgreSQL enum.
2. Add all new columns (`drug_class`, `drug_name`, `status`, `resolution_type`, `resolution_note`, `resolved_by_user_id`, `resolved_at`, `sla_breached`) to the existing table.
3. Create new enum types (`alert_status_enum`, `alert_resolution_type_enum`) before the `ADD COLUMN` statements.
4. Populate `status = 'ACTIVE'` for all existing rows (backfill).
5. Be fully reversible in `downgrade()`.

**Design references:**
- US-032 AC Scenarios 1, 2, 3 — all new columns required
- design.md §4.1 — Alembic for version-controlled schema migrations
- ADR-003 — Cloud SQL PostgreSQL 15; append-only audit log policy

```python
"""extend_pharmacist_alerts_high_risk_drug_class

Revision ID: <auto>
Revises: <previous_revision>
Create Date: 2026-07-16

Extends the pharmacist_alerts table to support HIGH_RISK_DRUG_CLASS alert type
and the pharmacist resolution workflow (US-032).
"""
from __future__ import annotations

from alembic import op
import sqlalchemy as sa
from sqlalchemy.dialects.postgresql import UUID


def upgrade() -> None:
    # 1. Extend alert_type_enum to include HIGH_RISK_DRUG_CLASS
    # PostgreSQL requires ALTER TYPE ... ADD VALUE outside a transaction.
    op.execute(
        "ALTER TYPE alert_type_enum ADD VALUE IF NOT EXISTS 'HIGH_RISK_DRUG_CLASS'"
    )

    # 2. Create new enum types
    alert_status_enum = sa.Enum(
        "ACTIVE", "RESOLVED", name="alert_status_enum"
    )
    alert_resolution_type_enum = sa.Enum(
        "REVIEWED_ACCEPTABLE",
        "DOSE_ADJUSTED",
        "DRUG_CHANGED",
        "DISCONTINUED",
        name="alert_resolution_type_enum",
    )
    alert_status_enum.create(op.get_bind(), checkfirst=True)
    alert_resolution_type_enum.create(op.get_bind(), checkfirst=True)

    # 3. Add new columns
    op.add_column(
        "pharmacist_alerts",
        sa.Column("drug_class", sa.String(64), nullable=True),
    )
    op.add_column(
        "pharmacist_alerts",
        sa.Column("drug_name", sa.String(255), nullable=True),
    )
    op.add_column(
        "pharmacist_alerts",
        sa.Column(
            "status",
            sa.Enum("ACTIVE", "RESOLVED", name="alert_status_enum"),
            nullable=False,
            server_default="ACTIVE",
        ),
    )
    op.add_column(
        "pharmacist_alerts",
        sa.Column(
            "resolution_type",
            sa.Enum(
                "REVIEWED_ACCEPTABLE",
                "DOSE_ADJUSTED",
                "DRUG_CHANGED",
                "DISCONTINUED",
                name="alert_resolution_type_enum",
            ),
            nullable=True,
        ),
    )
    op.add_column(
        "pharmacist_alerts",
        sa.Column("resolution_note", sa.Text(), nullable=True),
    )
    op.add_column(
        "pharmacist_alerts",
        sa.Column(
            "resolved_by_user_id",
            UUID(as_uuid=True),
            sa.ForeignKey("users.id", ondelete="SET NULL"),
            nullable=True,
        ),
    )
    op.add_column(
        "pharmacist_alerts",
        sa.Column(
            "resolved_at",
            sa.DateTime(timezone=True),
            nullable=True,
        ),
    )
    op.add_column(
        "pharmacist_alerts",
        sa.Column(
            "sla_breached",
            sa.Boolean(),
            nullable=False,
            server_default=sa.false(),
        ),
    )

    # 4. Add indexes for query performance
    op.create_index(
        "ix_pharmacist_alerts_drug_class",
        "pharmacist_alerts",
        ["drug_class"],
    )
    op.create_index(
        "ix_pharmacist_alerts_status",
        "pharmacist_alerts",
        ["status"],
    )
    op.create_index(
        "ix_pharmacist_alerts_resolved_by_user_id",
        "pharmacist_alerts",
        ["resolved_by_user_id"],
    )

    # 5. Backfill existing rows: set status = 'ACTIVE'
    op.execute("UPDATE pharmacist_alerts SET status = 'ACTIVE' WHERE status IS NULL")


def downgrade() -> None:
    op.drop_index("ix_pharmacist_alerts_resolved_by_user_id", "pharmacist_alerts")
    op.drop_index("ix_pharmacist_alerts_status", "pharmacist_alerts")
    op.drop_index("ix_pharmacist_alerts_drug_class", "pharmacist_alerts")

    for col in [
        "sla_breached",
        "resolved_at",
        "resolved_by_user_id",
        "resolution_note",
        "resolution_type",
        "status",
        "drug_name",
        "drug_class",
    ]:
        op.drop_column("pharmacist_alerts", col)

    op.execute("DROP TYPE IF EXISTS alert_resolution_type_enum")
    op.execute("DROP TYPE IF EXISTS alert_status_enum")
    # Note: PostgreSQL does not support removing a value from an enum type.
    # alert_type_enum HIGH_RISK_DRUG_CLASS value cannot be removed without
    # dropping and recreating the enum — acceptable risk for downgrade path.
```
